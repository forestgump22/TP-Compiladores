grammar Preposicion;


// Parser
program
  : stmt+ EOF ;


stmt
  : 'if' exp '{' stmt '}' 'else' '{' stmt '}'?  
    | exp ';'                              
    ;


exp
  : ID
  |NUM
  |exp '&&' exp
  | exp '||' exp  
  | exp '==' exp
  | exp '!=' exp  
  | '(' exp ')'    
  ;


// Lexer


OR: '||';
AND: '&&';
IG: '==';
DIF: '!=';
ID  : [a-zA-Z_][a-zA-Z_0-9]* ;
NUM : [0-9]+('.'[0-9]+)? ;
CMNT: '#' [ a-zA-Z_0-9.:()]* '\n' -> skip ;
WS  : [ \n\t\r]+ -> skip ;
